7   package com.varmateo.yawg.util;
8   
9   import java.io.IOException;
10  import java.io.Reader;
11  import java.util.Collections;
12  import java.util.Map;
13  
14  import com.esotericsoftware.yamlbeans.YamlReader;
15  
16  import com.varmateo.yawg.util.SimpleMap;
17  
18  
19  /**
20   * A simple YAML parser.
21   */
22  public final class YamlParser {
23  
24  
25      /**
26       * Parses YAML contants from the given reader.
27       *
28       * @param reader Source of YAML contents to be read.
29       *
30       * @return A map with the results of parsing the given YAML
31       * content.
32       *
33       * @exception IOException Thrown if there were any problems
34       * reading from the reader, or if the YAML contents are invalid.
35       */
36      public SimpleMap parse(final Reader reader)
37              throws IOException {
38  
39          final Map<String,Object> resultMap;
40          YamlReader yamlReader = new YamlReader(reader);
41          Object yamlObj = yamlReader.read();
42  
43          if ( (yamlObj!=null) && (yamlObj instanceof Map) ) {
44              @SuppressWarnings("unchecked")
45              Map<String,Object> map = (Map<String,Object>)yamlObj;
46              resultMap = map;
47          } else {
48              // The contents of the YAML file are invalid.
49              resultMap = Collections.emptyMap();
50          }
51  
52          return new SimpleMap(resultMap);
53      }
54  
55  
56  }
