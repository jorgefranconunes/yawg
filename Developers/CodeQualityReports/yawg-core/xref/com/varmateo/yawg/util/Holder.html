7   package com.varmateo.yawg.util;
8   
9   import java.util.function.Supplier;
10  
11  
12  /**
13   * A memoizer for a single object.
14   *
15   * @param <T> The type of objects the holder will hold.
16   */
17  public final class Holder<T> {
18  
19  
20      private Supplier<T> _getter  = null;
21      private boolean     _isBuilt = false;
22  
23  
24      /**
25       *
26       */
27      private Holder(final Supplier<T> builder) {
28  
29          _getter = () -> buildAndCache(builder);
30      }
31  
32  
33      /**
34       *
35       */
36      private T buildAndCache(final Supplier<T> builder) {
37  
38          synchronized ( this ) {
39              if ( !_isBuilt ) {
40                  T payload = builder.get();
41  
42                  _getter  = () -> payload;
43                  _isBuilt = true;
44              }
45          }
46  
47          return _getter.get();
48      }
49  
50  
51      /**
52       * Retrieves the object contained in this holder.
53       *
54       * <p>The first time this method is called, the
55       * <code>Supplier</code> passed to the <code>{@link
56       * #of(Supplier<T>)}</code> method will be invoked to fetch the
57       * actual object. All following calls will then always return that
58       * same object.</p>
59       *
60       * @return The object obtained from the initial
61       * <code>Supplier</code> this holder was constructd from.
62       */
63      public T get() {
64  
65          return _getter.get();
66      }
67  
68  
69      /**
70       * Creates a new holder whose content will be retrieved from the
71       * given <code>Supplier</code>.
72       *
73       * @param builder The supplier from which the object contained in
74       * the holder will be obtained.
75       *
76       * @return A newly created holder.
77       *
78       * @param <T> The type of object the returned holder will hold.
79       */
80      public static <T> Holder<T> of(final Supplier<T> builder) {
81  
82          return new Holder<T>(builder);
83      }
84  
85  
86  }
