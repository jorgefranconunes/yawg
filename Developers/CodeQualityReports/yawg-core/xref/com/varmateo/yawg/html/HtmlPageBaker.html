24  /**
25   * A <code>Baker</code> that transforms HTML files into other HTML
26   * files.
27   */
28  public final class HtmlPageBaker
29          implements PageBaker {
30  
31  
32      private static final String NAME = "html";
33  
34      private static final Pattern RE_HTML = Pattern.compile(".*\\.html$");
35  
36      private static final String TARGET_EXTENSION = ".html";
37  
38  
39      private HtmlPageBaker() {
40          // Nothing to do.
41      }
42  
43  
44      /**
45       *
46       */
47      public static PageBaker create() {
48  
49          return new HtmlPageBaker();
50      }
51  
52  
53      /**
54       * {@inheritDoc}
55       */
56      @Override
57      public String shortName() {
58  
59          return NAME;
60      }
61  
62  
63      /**
64       * Checks if the given file name has one of the known extensions.
65       *
66       * <p>The following extensions will be allowed:</p>
67       *
68       * <ul>
69       *   <li>.adoc</li>
70       * </ul>
71       *
72       * @return True if the given file name has one of the allowed
73       * extensions.
74       */
75      @Override
76      public boolean isBakeable(final Path path) {
77  
78          return FileUtils.isNameMatch(path, RE_HTML);
79      }
80  
81  
82      /**
83       * Converts the given text file in HTML format into another HTML
84       * file processed through the template engine.
85       *
86       * <p>The target directory must already exist. Otherwise an
87       * exception will be thrown.</p>
88       *
89       * @param sourcePath The file to be baked.
90       *
91       * @param context Provides the template for generating the target
92       * document. If no template is provided, then the source document
93       * is just copied to the target directory.
94       *
95       * @param targetDir The directory where the baked file will be
96       * copied to.
97       *
98       * @exception YawgException Thrown if the baking failed for
99       * whatever reason.
100      */
101     @Override
102     public void bake(
103             final Path sourcePath,
104             final PageContext context,
105             final Path targetDir)
106             throws YawgException {
107 
108         try {
109             doBake(sourcePath, context, targetDir);
110         } catch ( IOException e ) {
111             Exceptions.raise(
112                     e,
113                     "Failed {0} on {1} - {2} - {3}",
114                     NAME,
115                     sourcePath,
116                     e.getClass().getSimpleName(),
117                     e.getMessage());
118         }
119     }
120 
121 
122     /**
123      *
124      */
125     private void doBake(
126             final Path sourcePath,
127             final PageContext context,
128             final Path targetDir)
129             throws IOException {
130 
131         final Path targetPath = getTargetPath(sourcePath, targetDir);
132         final Optional<Template> template = context.templateFor(sourcePath);
133 
134         if ( template.isPresent() ) {
135             doBakeWithTemplate(sourcePath, context, targetPath, template.get());
136         } else {
137             doBakeWithoutTemplate(sourcePath, targetPath);
138         }
139     }
140 
141 
142     /**
143      *
144      */
145     private Path getTargetPath(
146             final Path sourcePath,
147             final Path targetDir) {
148 
149         final String sourceBasename = FileUtils.basename(sourcePath);
150         final String targetName = sourceBasename + TARGET_EXTENSION;
151 
152         return targetDir.resolve(targetName);
153     }
154 
155 
156     /**
157      * Baking without a template just copies the source HTML file to
158      * the target location.
159      */
160     private void doBakeWithoutTemplate(
161             final Path sourcePath,
162             final Path targetPath)
163             throws IOException {
164 
165         FileUtils.copy(sourcePath, targetPath);
166     }
167 
168 
169     /**
170      *
171      */
172     private void doBakeWithTemplate(
173             final Path sourcePath,
174             final PageContext context,
175             final Path targetPath,
176             final Template template)
177             throws IOException {
178 
179         final HtmlBakerDataModelBuilder#HtmlBakerDataModelBuilder">HtmlBakerDataModelBuilder modelBuilder = new HtmlBakerDataModelBuilder();
180         final TemplateDataModel dataModel = modelBuilder.build(sourcePath, targetPath, context);
181 
182         FileUtils.newWriter(
183                 targetPath,
184                 writer -> template.process(dataModel, writer));
185     }
