7   package com.varmateo.yawg.html;
8   
9   import java.nio.file.Path;
10  import java.util.Optional;
11  import java.util.function.Function;
12  
13  import io.vavr.control.Try;
14  import org.jsoup.Jsoup;
15  import org.jsoup.nodes.Document;
16  import org.jsoup.nodes.Element;
17  
18  import com.varmateo.yawg.spi.PageContext;
19  import com.varmateo.yawg.spi.TemplateContext;
20  import com.varmateo.yawg.spi.TemplateContextBuilder;
21  import com.varmateo.yawg.util.FileUtils;
22  
23  
24  /**
25   * Creates a <code>TemplateContext</code> from an input HTML
26   * file.
27   */
28  final class HtmlTemplateContext {
29  
30  
31      private static final String TAG_BODY = "body";
32      private static final String TAG_TITLE = "title";
33  
34  
35      private HtmlTemplateContext() {
36          // Nothing to do.
37      }
38  
39  
40      /**
41       *
42       */
43      public static Try<TemplateContext> create(
44              final Path sourcePath,
45              final Path targetPath,
46              final PageContext context) {
47  
48          return parse(sourcePath)
49                  .map(buildTemplateContext(sourcePath, targetPath, context));
50      }
51  
52  
53      private static Try<Document> parse(final Path sourcePath) {
54  
55          return Try.of(() -> Jsoup.parse(sourcePath.toFile(), null))
56                  .recoverWith(HtmlPageBakerException.parseFailureTry(sourcePath));
57      }
58  
59  
60      private static Function<Document, TemplateContext> buildTemplateContext(
61              final Path sourcePath,
62              final Path targetPath,
63              final PageContext context) {
64  
65          return (Document document) -> buildTemplateContext(
66                  sourcePath, targetPath, context, document);
67      }
68  
69  
70      private static TemplateContext buildTemplateContext(
71              final Path sourcePath,
72              final Path targetPath,
73              final PageContext context,
74              final Document document) {
75  
76          final String body = Optional.ofNullable(document.getElementsByTag(TAG_BODY).first())
77                  .map(Element::html)
78                  .orElse("");
79          final String pageUrl = context.dirUrl() + "/" + targetPath.getFileName();
80          final String title = Optional.ofNullable(document.getElementsByTag(TAG_TITLE).first())
81                  .map(Element::text)
82                  .orElseGet(() -> FileUtils.basename(sourcePath));
83  
84          return TemplateContextBuilder.create()
85                  .title(title)
86                  .body(body)
87                  .pageUrl(pageUrl)
88                  .rootRelativeUrl(context.rootRelativeUrl())
89                  .pageVars(context.pageVars())
90                  .bakeId(context.bakeId())
91                  .build();
92      }
93  
94  
95  }
