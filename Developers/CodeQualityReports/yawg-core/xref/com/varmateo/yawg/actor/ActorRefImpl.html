7   package com.varmateo.yawg.actor;

9   import java.util.concurrent.Executor;
10  import java.lang.reflect.InvocationHandler;
11  import java.lang.reflect.InvocationTargetException;
12  import java.lang.reflect.Method;
13  import java.lang.reflect.Proxy;
14  import java.util.concurrent.CompletableFuture;
15  import java.util.concurrent.CompletionException;
16  
17  import com.varmateo.yawg.actor.ActorRef;
18  import com.varmateo.yawg.actor.SequentialExecutor;
19  
20  
21  /**
22   * An internal implementation of <code>ActorRef</code> intended to be
23   * used solely by the <code>ActorSystem</code> implementation.
24   */
25  /* default */ final class ActorRefImpl<T>
26          implements ActorRef<T> {
27  
28  
29      private final Executor _sequentialExecutor;
30      private final ProxyActorInvocationHandler _proxyInvocationHandler;
31      private final T _actorSelf;
32  
33  
34      /**
35       *
36       * @param actorType The class object of the interface the actor
37       * instance implements.
38       *
39       * @param executor Used for executing actors methods.
40       */
41      /* default */ ActorRefImpl(
42              final Class<T> actorType,
43              final Executor executor) {
44  
45          Executor sequentialExecutor = new SequentialExecutor(executor);
46          ProxyActorInvocationHandler proxyInvocationHandler =
47                  new ProxyActorInvocationHandler(sequentialExecutor);
48  
49          @SuppressWarnings("unchecked")
50          T actorSelf =
51                  (T)Proxy.newProxyInstance(
52                          actorType.getClassLoader(),
53                          new Class<?>[] { actorType },
54                          proxyInvocationHandler);
55  
56          _sequentialExecutor = sequentialExecutor;
57          _proxyInvocationHandler = proxyInvocationHandler;
58          _actorSelf = actorSelf;
59      }
60  
61  
62      /**
63       * Completes the initialization of the <code>ActorRef</code>. From
64       * here on the object returned by the <code>{@link #self()}</code>
65       * method will perform as desired.
66       *
67       * @param actorCore The object being wrapped by this
68       * <code>ActorRef</code>.
69       */
70      /* default */ void setActorCore(final T actorCore) {
71  
72          _proxyInvocationHandler.setActorCore(actorCore);
73      }
74  
75  
76      /**
77       * {@inheritDoc}
78       */
79      @Override
80      public void perform(final Runnable action) {
81  
82          _sequentialExecutor.execute(action);
83      }
84  
85  
86      /**
87       * {@inheritDoc}
88       */
89      @Override
90      public T self() {
91  
92          return _actorSelf;
93      }
94  
95  
96      /**
97       *
98       */
99      private static final class ProxyActorInvocationHandler
100             implements InvocationHandler {
101 
102 
103         private Object _actorCore;
104         private final Executor _executor;
105 
106 
107         /**
108          *
109          */
110         ProxyActorInvocationHandler(final Executor executor) {
111 
112             _executor = executor;
113         }
114 
115 
116         /**
117          *
118          */
119         public void setActorCore(final Object actorCore) {
120 
121             _actorCore = actorCore;
122         }
123 
124 
125         /**
126          *
127          */
128         @Override
129         public Object invoke(
130                 final Object proxy,
131                 final Method method,
132                 final Object[] args)
133                 throws Throwable {
134 
135             if ( method.getReturnType() == Void.TYPE ) {
136                 return invokeVoidMethod(method, args);
137             } else {
138                 return invokeNonVoidMethod(method, args);
139             }
140         }
141 
142 
143         /**
144          *
145          */
146         private Object invokeVoidMethod(
147                 final Method method,
148                 final Object[] args) {
149 
150             Runnable task = () -> doInvokeVoidMethod(_actorCore, method, args);
151 
152             _executor.execute(task);
153 
154             return null;
155         }
156 
157 
158         /**
159          *
160          */
161         private static void doInvokeVoidMethod(
162                 final Object actorCore,
163                 final Method method,
164                 final Object[] args) {
165 
166             try {
167                 method.invoke(actorCore, args);
168             } catch ( IllegalAccessException e ) {
169                 throw new RuntimeException(e);
170             } catch ( InvocationTargetException e ) {
171                 Throwable cause = e.getCause();
172                 if ( cause instanceof RuntimeException ) {
173                     throw (RuntimeException)cause;
174                 } else {
175                     throw new RuntimeException(cause);
176                 }
177             }
178         }
179 
180 
181         /**
182          *
183          */
184         private Object invokeNonVoidMethod(
185                 final Method method,
186                 final Object[] args)
187                 throws Throwable {
188 
189             CompletableFuture<Object> future = new CompletableFuture<>();
190             Runnable task =
191                     () -> doInvokeNonVoidMethod(_actorCore, method, args,future);
192 
193             _executor.execute(task);
194 
195             final Object result;
196 
197             try {
198                 result = future.join();
199             } catch ( CompletionException e ) {
200                 Throwable cause = e.getCause();
201                 throw cause;
202             }
203 
204             return result;
205         }
206 
207 
208         /**
209          *
210          */
211         private static void doInvokeNonVoidMethod(
212                 final Object actorCore,
213                 final Method method,
214                 final Object[] args,
215                 final CompletableFuture<Object> future) {
216 
217             try {
218                 Object result = method.invoke(actorCore, args);
219                 future.complete(result);
220             } catch ( IllegalAccessException e ) {
221                 future.completeExceptionally(new RuntimeException(e));
222             } catch ( InvocationTargetException e ) {
223                 future.completeExceptionally(e.getCause());
224             }
225         }
226 
227 
228     }
