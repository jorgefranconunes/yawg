7   package com.varmateo.yawg.actor;
8   
9   import java.util.concurrent.Executor;
10  
11  import com.varmateo.yawg.actor.ActorFactory;
12  import com.varmateo.yawg.actor.ActorRefImpl;
13  
14  
15  /**
16   * Manages a set of actors.
17   *
18   * <p>The actors managed by one given <code>ActorSystem</code> have
19   * their methods executed by one of the threads in the
20   * <code>Executor</code> specified at constructor time.</p>
21   */
22  public final class ActorSystem {
23  
24  
25      private final Executor _executor;
26  
27  
28      /**
29       * @param executor Used for executing actor methods.
30       */
31      public ActorSystem(final Executor executor) {
32  
33          _executor = executor;
34      }
35  
36  
37      /**
38       * Creates a new actor.
39       *
40       * <p>A new actor instance is created through the provided
41       * <code>{@link ActorFactory}</code>. A proxy for that actor
42       * instance will then be returned. The actor proxy is the visible
43       * face of the actor for the outside world. The actor proxy,
44       * working within the actor system, has the responsability to
45       * ensure the actor methods are called in sequence, and never
46       * concurrently.</p>
47       *
48       * @param <T> Type of objects to be created. This will have to be
49       * an interface type.
50       *
51       * @param actorFactory Used for creating an actual actor
52       * instance.
53       *
54       * @param actorType The class object of the interface the actor
55       * instance implements. This will also be the type of the returned
56       * proxy actor.
57       *
58       * @return An object implementing the given <code>actorType</code>
59       * interface. That object is a proxy for the actor instance
60       * created with the <code>ActorFactory</code>.
61       */
62      public <T> T createActor(
63              final ActorFactory<T> actorFactory,
64              final Class<T> actorType) {
65  
66          ActorRefImpl<T> actorRef = new ActorRefImpl<>(actorType, _executor);
67          T actorCore = actorFactory.newActor(actorRef);
68  
69          actorRef.setActorCore(actorCore);
70  
71          return actorRef.self();
72      }
73  
74  }
