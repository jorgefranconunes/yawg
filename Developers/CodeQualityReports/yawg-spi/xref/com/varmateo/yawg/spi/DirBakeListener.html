9   import com.varmateo.yawg.api.YawgException;
10  import com.varmateo.yawg.spi.PageContext;
11  import com.varmateo.yawg.spi.PageVars;
12  
13  
14  /**
15   * Interface for objects that want to be notified when one given
16   * directory is baked during the baking procedure.
17   */
18  public interface DirBakeListener {
19  
20  
21      /**
22       * Called when a given directory is being baked during the baking
23       * procedure.
24       *
25       * <p>The listener has the opportunity to update the set of
26       * template variables that will be used by the baker when baking
27       * the files in the directory.</p>
28       *
29       * @param context Data on the directory being baked.
30       *
31       * @return An updated set of template variables to be used by the
32       * baker. If the listener does not want to make any change to the
33       * template variables it should return the same
34       * <code>PageVars</code> instance contained in the given
35       * <code>PageContext</code>.
36       *
37       * @throws YawgException If there were any problems performing its
38       * activities.
39       */
40      PageVars onDirBake(PageContext context)
41              throws YawgException;
42  
43  
44  }
