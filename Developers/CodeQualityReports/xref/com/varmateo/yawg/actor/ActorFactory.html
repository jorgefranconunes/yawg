7   package com.varmateo.yawg.actor;

9   import com.varmateo.yawg.actor.ActorRef;
10  
11  
12  /**
13   * Contract for a factory of actors.
14   *
15   * <p>An <code>ActorFactory</code> is required for creating an actor
16   * instance in the actor system through <code>{@link
17   * ActorSystem#createActor(Class,ActorFactory)}</code>.
18   */
19  @FunctionalInterface
20  public interface ActorFactory<T> {
21  
22  
23      /**
24       * Creates a new actor instance.
25       *
26       * <p>The factory method will receive the <code>{@link
27       * ActorRef}</code> with which the new actor instance will be
28       * associated to.</code>
29       *
30       * @param actorSelf The actor reference that the returned actor
31       * instance will be associated to.
32       *
33       * @return A new actor instance.
34       */
35      T newActor(ActorRef<T> actorSelf);
36  
37  }
