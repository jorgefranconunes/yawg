7   package com.varmateo.yawg.html;
8   
9   import java.io.IOException;
10  import java.nio.file.Path;
11  
12  import com.varmateo.yawg.api.YawgException;
13  
14  
15  /**
16   *
17   */
18  public final class HtmlPageBakerException
19          extends YawgException {
20  
21  
22      private HtmlPageBakerException(
23              final String msg,
24              final Throwable cause) {
25  
26          super(msg, cause);
27      }
28  
29  
30      /**
31       *
32       */
33      public static HtmlPageBakerException bakeFailure(
34              final Path sourcePath,
35              final Path targetDir,
36              final IOException cause) {
37  
38          final String msg = String.format(
39                  "Failed to bake HTML file \"%s\" into directory \"%s\"- %s - %s",
40                  sourcePath,
41                  targetDir,
42                  cause.getClass().getName(),
43                  cause.getMessage());
44  
45          return new HtmlPageBakerException(msg, cause);
46      }
47  
48  }
