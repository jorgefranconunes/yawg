7   package com.varmateo.yawg.core;
8   
9   import java.nio.file.Path;
10  import java.util.function.Function;
11  import java.util.function.Supplier;
12  
13  import io.vavr.Lazy;
14  import io.vavr.collection.List;
15  import io.vavr.collection.Seq;
16  import io.vavr.control.Option;
17  
18  import com.varmateo.yawg.api.SiteBaker;
19  import com.varmateo.yawg.spi.PageBaker;
20  import com.varmateo.yawg.spi.DirBakeListener;
21  import com.varmateo.yawg.spi.TemplateService;
22  import com.varmateo.yawg.spi.TemplateServiceFactory;
23  
24  
25  /**
26   * Module for all the objects required by the main baker.
27   */
28  /* default */ final class DefaultSiteBakerModule {
29  
30  
31      private final Lazy<SiteBaker> _siteBaker = Lazy.of(this::newSiteBaker);
32      private final Lazy<FileBaker> _fileBaker = Lazy.of(this::newFileBaker);
33      private final Lazy<DirBakeOptionsDao> _dirBakerOptionsDao = Lazy.of(this::newDirBakeOptionsDao);
34      private final Lazy<PageBaker> _defaultPageBaker = Lazy.of(this::newDefaultPageBaker);
35  
36      private final Supplier<Seq<PageBaker>> _pageBakers;
37      private final Supplier<Seq<TemplateServiceFactory>> _templateServiceFactories;
38      private final Supplier<Seq<DirBakeListener>> _dirBakeListeners;
39  
40  
41      private DefaultSiteBakerModule(
42              final Supplier<Seq<PageBaker>> pageBakers,
43              final Supplier<Seq<TemplateServiceFactory>> templateServiceFactories,
44              final Supplier<Seq<DirBakeListener>> dirBakeListeners) {
45  
46          _pageBakers = pageBakers;
47          _templateServiceFactories = templateServiceFactories;
48          _dirBakeListeners = dirBakeListeners;
49      }
50  
51  
52      /**
53       *
54       */
55      public static DefaultSiteBakerModule create(
56              final Supplier<Seq<PageBaker>> pageBakers,
57              final Supplier<Seq<TemplateServiceFactory>> templateServiceFactories,
58              final Supplier<Seq<DirBakeListener>> dirBakeListeners) {
59  
60          return new DefaultSiteBakerModule(pageBakers, templateServiceFactories, dirBakeListeners);
61      }
62  
63  
64      /**
65       * @return The baker object.
66       */
67      public SiteBaker siteBaker() {
68  
69          return _siteBaker.get();
70      }
71  
72  
73      private SiteBaker newSiteBaker() {
74  
75          final AssetsCopier assetsCopier = AssetsCopier.create();
76          final Function<Option<Path>, DirBaker> dirBakerFactory = this::newDirBaker;
77  
78          return DefaultSiteBaker.create(assetsCopier, dirBakerFactory);
79      }
80  
81  
82      /**
83       *
84       */
85      private DirBaker newDirBaker(final Option<Path> templatesDir) {
86  
87          final Function<Path, Seq<TemplateService>> createTemplateServices =
88                  path -> _templateServiceFactories.get().map(f -> f.newTemplateService(path));
89          final Seq<TemplateService> allTemplateServices = templatesDir
90                  .map(createTemplateServices)
91                  .getOrElse(List::empty);
92          final TemplateService templateService =
93                  new CollectiveTemplateService(allTemplateServices);
94          final DirBakeListener dirBakeListener =
95                  new CollectiveDirBakeListener(_dirBakeListeners.get());
96  
97          return new DirBaker(
98                  _fileBaker.get(),
99                  templateService,
100                 _dirBakerOptionsDao.get(),
101                 dirBakeListener);
102     }
103 
104 
105     /**
106      *
107      */
108     private FileBaker newFileBaker() {
109 
110         final Seq<PageBaker> bakers = _pageBakers.get();
111         final PageBaker defaultBaker = _defaultPageBaker.get();
112 
113         return new FileBaker(bakers, defaultBaker);
114     }
115 
116 
117     /**
118      *
119      */
120     private DirBakeOptionsDao newDirBakeOptionsDao() {
121 
122         return new DirBakeOptionsDao();
123     }
124 
125 
126     /**
127      *
128      */
129     private PageBaker newDefaultPageBaker() {
130 
131         return new CopyPageBaker();
132     }
133

135 }
