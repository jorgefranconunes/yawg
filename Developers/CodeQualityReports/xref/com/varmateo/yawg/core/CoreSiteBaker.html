21  import com.varmateo.yawg.logging.Logs;

24  import com.varmateo.yawg.util.Results;

26  
27  /**
28   * 
29   */
30  /* default */ final class CoreSiteBaker
31          implements SiteBaker {
32  
33  
34      private static final String DEFAULT_TEMPLATE_NAME = "default.ftlh";
35  
36      private final Log _log;
37      private final DirBaker _dirBaker;
38  
39  
40      private CoreSiteBaker(
41              final Log log,
42              final DirBaker dirBaker) {
43  
44          _log = log;
45          _dirBaker = dirBaker;
46      }
47  
48  
49      /**
50       *
51       */
52      public static SiteBaker create(final DirBaker dirBaker) {
53  
54          final Log log = LogFactory.createFor(CoreSiteBaker.class);
55  
56          return new CoreSiteBaker(log, dirBaker);
57      }
58  
59  
60      /**
61       *
62       */
63      @Override
64      public Result<Void> bake(final BakeOptions options) {
65  
66          return Results.fromTry(
67                  Try.run(
68                          () -> Logs.logDuration(
69                                  _log,
70                                  "baking",
71                                  () -> doBake(options, _dirBaker))));
72      }
73  
74  
75      /**
76       *
77       */
78      private void doBake(
79              final BakeOptions options,
80              final DirBaker dirBaker)
81              throws YawgException {
82  
83          final Path sourceDir = options.sourceDir();
84          final Path targetDir = options.targetDir();
85          final PageVars externalPageVars = mapToPageVars(options.externalPageVars());
86          final DirBakeOptions dirBakeOptions = DirBakeOptions.builder()
87                  .templateName(DEFAULT_TEMPLATE_NAME)
88                  .pageVars(externalPageVars)
89                  .build();
90  
91          dirBaker.bakeDirectory(sourceDir, targetDir, dirBakeOptions);
92      }
93  
94  
95      /**
96       *
97       */
98      private static PageVars mapToPageVars(
99              final Map<String,Object> pageVarsMap) {
100 
101         final PageVarsBuilder builder = PageVarsBuilder.create();
102         pageVarsMap.entrySet().stream()
103                 .forEach(x -> builder.addVar(x.getKey(), x.getValue()));
104 
105         return builder.build();
106     }
107 
108 
109 }
