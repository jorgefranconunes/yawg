26  /**
27   * Baker for one specific site.
28   */
29  /* default */ final class SingleSiteBaker {
30  
31  
32      private final Log _log;
33      private final AssetsCopier _assetsCopier;
34      private final SourceTreeBaker _sourceTreeBaker;
35  
36  
37      /**
38       * @param options All the parameters needed for performing a bake.
39       */
40      SingleSiteBaker(final DirBaker dirBaker) {
41  
42          _log = LogFactory.createFor(SingleSiteBaker.class);
43          _assetsCopier = AssetsCopier.create();
44          _sourceTreeBaker = SourceTreeBaker.create(dirBaker);
45      }
46  
47  
48      /**
49       * Performs the baking of one directory tree into another.
50       *
51       * @exception YawgException Thrown if the baking could not be
52       * completed for whatever reason.
53       */
54      public void bake(final BakeOptions options)
55              throws YawgException {
56  
57          Logs.logDuration(
58                  _log,
59                  "baking",
60                  () -> doBake(options));
61      }
62  
63  
64      /**
65       *
66       */
67      private void doBake(final BakeOptions options)
68              throws YawgException {
69  
70          Option.ofOptional(options.assetsDir())
71                  .peek(assetsDir -> _assetsCopier.copy(assetsDir, options.targetDir()))
72                  .onEmpty(() -> _log.debug("No assets directory to copy"));
73  
74          final PageVars externalPageVars = mapToPageVars(options.externalPageVars());
75  
76          _sourceTreeBaker.bake(options.sourceDir(), options.targetDir(), externalPageVars);
77      }
78  
79  
80      /**
81       *
82       */
83      private static PageVars mapToPageVars(
84              final Map<String,Object> pageVarsMap) {
85  
86          final PageVarsBuilder builder = PageVarsBuilder.create();
87          pageVarsMap.entrySet().stream()
88                  .forEach(x -> builder.addVar(x.getKey(), x.getValue()));
89  
90          return builder.build();
91      }
92  
93  
94  }
