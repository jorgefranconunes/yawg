7   package com.varmateo.yawg.core;
8   
9   import java.nio.file.Path;
10  import java.util.Optional;
11  
12  import com.varmateo.yawg.api.YawgException;
13  import com.varmateo.yawg.api.YawgInfo;
14  import com.varmateo.yawg.core.DirBaker;
15  import com.varmateo.yawg.logging.Log;
16  import com.varmateo.yawg.logging.LogWithUtils;
17  import com.varmateo.yawg.spi.SiteBakerConf;
18  
19  
20  /**
21   * Baker for one specific site.
22   */
23  /* package private */ final class SingleSiteBaker {
24  
25  
26      private final LogWithUtils _log;
27      private final SiteBakerConf _conf;
28      private final AssetsCopier _assetsCopier;
29      private final SourceTreeBaker _sourceTreeBaker;
30  
31  
32      /**
33       * @param conf All the parameters needed for performing a bake.
34       */
35      SingleSiteBaker(
36              final Log log,
37              final SiteBakerConf conf,
38              final DirBaker dirBaker) {
39  
40          _log = LogWithUtils.from(log);
41          _conf = conf;
42          _assetsCopier = new AssetsCopier(
43                  _log,
44                  conf.getAssetsDir(),
45                  conf.getTargetDir());
46          _sourceTreeBaker = new SourceTreeBaker(
47                  _log,
48                  conf.getSourceDir(),
49                  conf.getTargetDir(),
50                  conf.getExternalPageVars(),
51                  dirBaker);
52      }
53  
54  
55      /**
56       * Performs the baking of one directory tree into another.
57       *
58       * @exception YawgException Thrown if the baking could not be
59       * completed for whatever reason.
60       */
61      public void bake()
62              throws YawgException {
63  
64          Path sourceDir = _conf.getSourceDir();
65          Path targetDir = _conf.getTargetDir();
66          Optional<Path> templatesDir = _conf.getTemplatesDir();
67          Optional<Path> assetsDir = _conf.getAssetsDir();
68  
69          _log.info("{0} {1}", YawgInfo.PRODUCT_NAME, YawgInfo.VERSION);
70          _log.info("    Source    : {0}", sourceDir);
71          _log.info("    Target    : {0}", targetDir);
72          _log.info(
73                  "    Templates : {0}",
74                  templatesDir.map(Path::toString).orElse("NONE"));
75          _log.info(
76                  "    Assets    : {0}",
77                  assetsDir.map(Path::toString).orElse("NONE"));
78  
79          _log.logDelay("bake", this::doBake);
80      }
81  
82  
83      /**
84       *
85       */
86      private void doBake()
87              throws YawgException {
88  
89          _assetsCopier.copy();
90          _sourceTreeBaker.bake();
91      }
92  
93  
94  }
