29  /**
30   * Command line interface for the baker utility.
31   */
32  public final class BakerCli {
33  
34  
35      private static final int EXIT_STATUS_OK = 0;
36      private static final int EXIT_STATUS_FAILURE = 1;
37  
38  
39      private final Log _log = LogFactory.createFor(BakerCli.class);
40  
41  
42      private BakerCli() {
43          // Nothing to do.
44      }
45  
46  
47      /**
48       *
49       */
50      public static BakerCli create() {
51  
52          return new BakerCli();
53      }
54  
55  
56      /**
57       * The utility entry point.
58       *
59       * @return The utility exit status. Zero means all went
60       * well. Non-zero means something failed.
61       */
62      public int run(final BakerCliRunOptions options) {
63  
64          final InfoPrinter infoPrinter = buildInfoPrinter(options.argv0, options.output);
65          final Try<Void> bakeResult = tryRun(infoPrinter, options.args)
66                  .onFailure(cause -> infoPrinter.printError(cause));
67  
68          final int exitStatus = bakeResult
69                  .map(x -> EXIT_STATUS_OK)
70                  .recover(x -> EXIT_STATUS_FAILURE)
71                  .get();
72  
73          return exitStatus;
74      }
75  
76  
77      /**
78       *
79       */
80      private static InfoPrinter buildInfoPrinter(
81              final String argv0,
82              final OutputStream output) {
83  
84          final boolean autoFlush = true;
85          final Writer stdoutWriter = new OutputStreamWriter(output, StandardCharsets.UTF_8);
86          final PrintWriter stdout = new PrintWriter(stdoutWriter, autoFlush);
87  
88          return InfoPrinter.builder()
89                  .argv0(argv0)
90                  .output(stdout)
91                  .build();
92      }
93  
94  
95      /**
96       *
97       */
98      private Try<Void> tryRun(
99              final InfoPrinter infoPrinter,
100             final String[] args) {
101 
102         return BakerCliAction.parse(args)
103                 .flatMap(action -> performAction(infoPrinter, action));
104     }
105 
106 
107     private Try<Void> performAction(
108             final InfoPrinter infoPrinter,
109             final BakerCliAction action) {
110 
111         return action.match(
112                 () -> Try.run(infoPrinter::printHelp),
113                 () -> Try.run(infoPrinter::printVersion),
114                 bake -> doBake(bake.options()));
115     }
116 
117 
118     private Try<Void> doBake(final BakerCliBakeOptions options) {
119 
120         logOptions(options);
121 
122         final SiteBaker siteBaker = options.templatesDir()
123                 .map(path -> DefaultSiteBaker.create(path))
124                 .getOrElse(() -> DefaultSiteBaker.create());
125 
126         final BakeOptions bakeOptions = BakeOptionsBuilder.create()
127                 .sourceDir(options.sourceDir())
128                 .targetDir(options.targetDir())
129                 .assetsDir(options.assetsDir().toJavaOptional())
130                 .putAllExternalPageVars(options.externalPageVars().toJavaMap())
131                 .build();
132 
133         final Result<Void> result = siteBaker.bake(bakeOptions);
134 
135         return Results.toTry(result)
136                 .recoverWith(cause -> Try.failure(CliException.bakeFailure(cause)));
137     }
138 
139 
140     private void logOptions(final BakerCliBakeOptions options) {
141 
142         final Path sourceDir = options.sourceDir();
143         final Path targetDir = options.targetDir();
144         final String templatesDir = options.templatesDir().map(Path::toString).getOrElse("NONE");
145         final String assetsDir = options.assetsDir().map(Path::toString).getOrElse("NONE");
146 
147         _log.info("{0} {1}", YawgInfo.PRODUCT_NAME, YawgInfo.VERSION);
148         _log.info("    Source    : {0}", sourceDir);
149         _log.info("    Target    : {0}", targetDir);
150         _log.info("    Templates : {0}", templatesDir);
151         _log.info("    Assets    : {0}", assetsDir);
152     }
153
