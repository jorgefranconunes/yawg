31  /**
32   * A <code>Baker</code> that translates text files in Asciidoc format
33   * into HTML files.
34   */
35  public final class AsciidoctorPageBaker
36          implements PageBaker {
37  
38  
39      private static final String NAME = "asciidoc";
40  
41      private static final Pattern RE_ADOC = Pattern.compile(".*\\.adoc$");
42  
43      private static final String TARGET_EXTENSION = ".html";
44  
45      private final Lazy<Asciidoctor> _asciidoctor =
46              Lazy.of(this::newAsciidoctor);
47  
48      private final Lazy<AsciidoctorBakerDataModelBuilder> _modelBuilder =
49              Lazy.of(this::newAsciidoctorBakerDataModelBuilder);
50  
51  
52      private AsciidoctorPageBaker() {
53          // Nothing to do.
54      }
55  
56  
57      /**
58       *
59       */
60      public static PageBaker create() {
61  
62          return new AsciidoctorPageBaker();
63      }
64  
65  
66      /**
67       * {@inheritDoc}
68       */
69      @Override
70      public String shortName() {
71  
72          return NAME;
73      }
74  
75  
76      /**
77       * Checks if the given file name has one of the known extensions.
78       *
79       * <p>The following extensions will be allowed:</p>
80       *
81       * <ul>
82       *   <li>.adoc</li>
83       * </ul>
84       *
85       * @return True if the given file name has one of the allowed
86       * extensions.
87       */
88      @Override
89      public boolean isBakeable(final Path path) {
90  
91          return FileUtils.isNameMatch(path, RE_ADOC);
92      }
93  
94  
95      /**
96       * Converts the given text file in Asciidoc format into an HTML
97       * file.
98       *
99       * <p>The target directory must already exist. Otherwise an
100      * exception will be thrown.</p>
101      *
102      * @param sourcePath The file to be baked.
103      *
104      * @param context Provides the template for generating the target
105      * document. If no template is provided, then the default
106      * AsciidoctorJ document generator will be used.
107      *
108      * @param targetDir The directory where the source file will be
109      * copied to.
110      *
111      * @exception YawgException Thrown if the copying failed for
112      * whatever reason.
113      */
114     @Override
115     public void bake(
116             final Path sourcePath,
117             final PageContext context,
118             final Path targetDir)
119             throws YawgException {
120 
121         try {
122             doBake(sourcePath, context, targetDir);
123         } catch ( AsciidoctorCoreException | IOException e ) {
124             Exceptions.raise(
125                     e,
126                     "Failed {0} on {1} - {2} - {3}",
127                     NAME,
128                     sourcePath,
129                     e.getClass().getSimpleName(),
130                     e.getMessage());
131         }
132     }
133 
134 
135     /**
136      *
137      */
138     private void doBake(
139             final Path sourcePath,
140             final PageContext context,
141             final Path targetDir)
142             throws AsciidoctorCoreException, IOException {
143 
144         final Path targetPath = getTargetPath(sourcePath, targetDir);
145         final Optional<Template> template = context.templateFor(sourcePath);
146 
147         if ( template.isPresent() ) {
148             doBakeWithTemplate(
149                     sourcePath,
150                     context,
151                     targetDir,
152                     targetPath,
153                     template.get());
154         } else {
155             doBakeWithoutTemplate(
156                     sourcePath,
157                     context,
158                     targetDir,
159                     targetPath);
160         }
161     }
162 
163 
164     /**
165      *
166      */
167     private Path getTargetPath(
168             final Path sourcePath,
169             final Path targetDir) {
170 
171         final String sourceBasename = FileUtils.basename(sourcePath);
172         final String targetName = sourceBasename + TARGET_EXTENSION;
173 
174         return targetDir.resolve(targetName);
175     }
176 
177 
178     /**
179      *
180      */
181     private void doBakeWithoutTemplate(
182             final Path sourcePath,
183             final PageContext context,
184             final Path targetDir,
185             final Path targetPath)
186             throws AsciidoctorCoreException {
187 
188         final OptionsBuilder options = AdocUtils.buildOptionsForBakeWithoutTemplate(
189                 sourcePath,
190                 targetDir,
191                 targetPath,
192                 context.pageVars());
193         final File sourceFile = sourcePath.toFile();
194 
195         _asciidoctor.get().convertFile(sourceFile, options);
196     }
197 
198 
199     /**
200      *
201      */
202     private void doBakeWithTemplate(
203             final Path sourcePath,
204             final PageContext context,
205             final Path targetDir,
206             final Path targetPath,
207             final Template template)
208             throws AsciidoctorCoreException, IOException {
209 
210         final TemplateDataModel dataModel = _modelBuilder.get().build(
211                 sourcePath, targetDir, targetPath, context);
212 
213         FileUtils.newWriter(
214                 targetPath,
215                 writer -> template.process(dataModel, writer));
216     }
217 
218 
219     /**
220      *
221      */
222     private Asciidoctor newAsciidoctor() {
223 
224         final Asciidoctor asciidoctor = Asciidoctor.Factory.create();
225 
226         // The following is required for supporting PlantUML diagrams
227         // (cf. http://asciidoctor.org/docs/asciidoctor-diagram/).
228         asciidoctor.requireLibrary("asciidoctor-diagram/plantuml");
229 
230         return asciidoctor;
231     }
232 
233 
234     /**
235      *
236      */
237     private AsciidoctorBakerDataModelBuilder newAsciidoctorBakerDataModelBuilder() {
238         return new AsciidoctorBakerDataModelBuilder(_asciidoctor.get());
239     }
