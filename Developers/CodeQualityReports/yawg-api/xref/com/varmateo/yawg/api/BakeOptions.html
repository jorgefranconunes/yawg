7   package com.varmateo.yawg.api;
8   
9   import java.nio.file.Path;
10  import java.util.Map;
11  import java.util.Optional;
12  
13  
14  /**
15   * Set of configuration parameters for baking a site with a
16   * <code>SiteBaker</code> instance.
17   */
18  public interface BakeOptions {
19  
20  
21      /**
22       * Path of directory containing the assets files. These files will
23       * be copied without any changes to the target directory.
24       *
25       * <p>An empty <code>Optional</code> means no assets will be
26       * copied.</p>
27       */
28      Optional<Path> assetsDir();
29  
30  
31      /**
32       * Path of directory containing the documents to be baked.
33       */
34      Path sourceDir();
35  
36  
37      /**
38       * Path of directory where the results of the baking will be
39       * stored.
40       */
41      Path targetDir();
42  
43  
44      /**
45       * Set of page variables provided externally.
46       *
47       * <p>Page variables are intended to be used by templates.</p>
48       *
49       * @return A read-only map containing the page variables. Keys in
50       * the map are the variable names.
51       */
52      Map<String, Object> externalPageVars();
53  
54  }
